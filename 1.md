import java.io.*;
import java.math.*;
import java.util.*;
public class GCD
{
 public static double gcd(double a, double h)
{
 double temp;
while (true) 
{
 temp = a % h;
if (temp == 0)
return h;
a = h;
h = temp;
}
}
public static void main(String[] args)
{
double p = 17;

 {

e++;
